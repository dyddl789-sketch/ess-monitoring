create table ess_member (
    member_id      number primary key, --회원 번호
    member_name    varchar2(30) not null, --회원 이름
    member_userid  varchar2(30) not null unique, --회원 아이디
    member_pw      varchar2(30) not null, --회원 비밀번호
    location       varchar2(50) --설치 지역
);

create table ess_monitoring (
    ess_id         number primary key, --측정 데이터 번호
    member_id      number not null, --어느 회원 집 데이터인지 연결
    device_name    varchar2(50) not null, --장비명
    voltage        number(10,2) not null, --전압
    current_val    number(10,2) not null, --전류
    soc            number(5,2) not null, --배터리 충전율
    status         varchar2(20) not null, --상태
    measured_at    date default sysdate not null, --측정 시간
    
    constraint fk_ess_member
        foreign key (member_id) references ess_member(member_id),
        
    constraint chk_soc_range
        check (soc between 0 and 100)
);

drop table ess_monitoring;
create sequence ess_member_seq
start with 1
increment by 1
nocache
nocycle;

create sequence ess_monitoring_seq
start with 1
increment by 1
nocache
nocycle;

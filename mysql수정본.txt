CREATE TABLE ess_member (
  member_id INT PRIMARY KEY AUTO_INCREMENT,
  member_name VARCHAR(30) NOT NULL,
  member_userid VARCHAR(30) NOT NULL UNIQUE,
  member_pw VARCHAR(100) NOT NULL,
  user_type VARCHAR(20) NOT NULL,
  phone VARCHAR(20),
  email VARCHAR(50),
  join_date DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE ess_device (
  device_id INT PRIMARY KEY AUTO_INCREMENT,
  member_id INT NOT NULL,
  device_name VARCHAR(50) NOT NULL,
  location VARCHAR(100),
  capacity_kw DECIMAL(10,2),
  device_type VARCHAR(20) NOT NULL,
  status VARCHAR(20) NOT NULL,
  install_date DATE,
  CONSTRAINT fk_device_member
    FOREIGN KEY (member_id) REFERENCES ess_member(member_id)
    ON DELETE CASCADE
);

CREATE TABLE monitoring (
  monitor_id INT PRIMARY KEY AUTO_INCREMENT,
  device_id INT NOT NULL,
  voltage DECIMAL(10,2) NOT NULL,
  current_a DECIMAL(10,2) NOT NULL,
  soc DECIMAL(5,2) NOT NULL,
  power_output DECIMAL(10,2),
  record_time DATETIME DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_monitoring_device
    FOREIGN KEY (device_id) REFERENCES ess_device(device_id)
    ON DELETE CASCADE,
  CONSTRAINT chk_soc_range
    CHECK (soc BETWEEN 0 AND 100)
);

CREATE TABLE energy_log (
  log_id INT PRIMARY KEY AUTO_INCREMENT,
  device_id INT NOT NULL,
  daily_kwh DECIMAL(10,2),
  monthly_kwh DECIMAL(10,2),
  cost DECIMAL(10,2),
  log_date DATE NOT NULL,
  CONSTRAINT fk_energylog_device
    FOREIGN KEY (device_id) REFERENCES ess_device(device_id)
    ON DELETE CASCADE
);

CREATE TABLE ess_alert (
  alert_id INT PRIMARY KEY AUTO_INCREMENT,
  device_id INT NOT NULL,
  alert_type VARCHAR(30) NOT NULL,
  alert_level VARCHAR(20) NOT NULL,
  message VARCHAR(200) NOT NULL,
  status VARCHAR(20) NOT NULL,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_alert_device
    FOREIGN KEY (device_id) REFERENCES ess_device(device_id)
    ON DELETE CASCADE
);
